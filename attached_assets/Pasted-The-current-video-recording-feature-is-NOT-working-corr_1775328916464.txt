async function uploadVideo(blob) {
  const file = new File([blob], "video.webm");

  const formData = new FormData();
  formData.append("file", file);

  const res = await fetch("/api/upload", {
    method: "POST",
    body: formData,
  });

  const data = await res.json();
  return data.url;
}
